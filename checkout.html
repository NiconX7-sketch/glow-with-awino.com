// PayPal Payment - CORRECTED async function processPayPal() { console.log('PayPal button clicked'); showStatus('Redirecting to PayPal...', 'info'); const currentCart = JSON.parse(sessionStorage.getItem('checkoutCart') || '[]'); const currentCurrency = localStorage.getItem('currency') || 'USD'; let currentTotal = 0; currentCart.forEach(item => { currentTotal += currentCurrency === 'USD' ? item.price_usd : item.price_kes; }); console.log('PayPal request:', { total: currentTotal, currency: currentCurrency }); const orderIdFromDb = await createOrder('paypal'); if (!orderIdFromDb) { showStatus('Failed to create order. Please try again.', 'error'); return; } currentOrderId = orderIdFromDb; try { const response = await fetch('/api/create-payment', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ method: 'paypal', amount: currentTotal, currency: currentCurrency, orderId: currentOrderId }) }); const data = await response.json(); console.log('PayPal API response:', data); // Check for approvalUrl (PayPal specific) if (data.approvalUrl) { console.log('Redirecting to PayPal URL:', data.approvalUrl); sessionStorage.setItem('pendingOrderId', currentOrderId); window.location.href = data.approvalUrl; } else { showStatus('Payment failed: No PayPal redirect URL received', 'error'); await updateOrderStatus(currentOrderId, 'failed'); } } catch (error) { console.error('PayPal error:', error); showStatus('Payment processing failed.
